TYPE SAFETY

          
Typed end-to-end. Validated at every boundary.

          
Zod schemas validate at runtime. TypeScript types catch bugs at build time. One source of truth — from CLI to database to UI.

// @inbox/contracts/src/ids.ts

export const MessageIdSchema = z.string()
  .regex(/^msg_[0-9A-Za-z]{24}$/)
  .brand<'MessageId'>();

export const ConversationIdSchema = z.string()
  .regex(/^cnv_[0-9A-Za-z]{24}$/)
  .brand<'ConversationId'>();

export const DeliveryIdSchema = z.string()
  .regex(/^dly_[0-9A-Za-z]{24}$/)
  .brand<'DeliveryId'>();

export type MessageId = z.infer<typeof MessageIdSchema>;
export type ConversationId = z.infer<typeof ConversationIdSchema>;
export type DeliveryId = z.infer<typeof DeliveryIdSchema>;



export const EngagementStateSchema = z.enum([
  'unread',
  'read',
  'acknowledged',
]);

export const VisibilityStateSchema = z.enum([
  'active',
  'hidden',
]);

export const UrgencySchema = z.enum([
  'normal',
  'urgent',
  'fyi',
]);



export const MessageEntitySchema = z.object({
  message_id: MessageIdSchema,
  conversation_id: ConversationIdSchema,
  sender_address: AddressStringSchema,
  subject: z.string().max(500),
  body: z.string().max(100_000),
  urgency: UrgencySchema,
  sent_at_ms: UnixMsSchema,
  parent_message_id: MessageIdSchema.nullable(),
});

export type MessageEntity = z.infer<typeof MessageEntitySchema>;



export const InboxListItemSchema = z.object({
  delivery_id: DeliveryIdSchema,
  message: MessageEntitySchema,
  engagement_state: EngagementStateSchema,
  visibility_state: VisibilityStateSchema,
  delivered_at_ms: UnixMsSchema,
  read_at_ms: UnixMsSchema.nullable(),
  acknowledged_at_ms: UnixMsSchema.nullable(),
});

export type InboxListItem = z.infer<typeof InboxListItemSchema>;

              API
              TypeScript / Hono server
            

            

              typescript
              

import { Hono } from 'hono';
import { SendRequestSchema } from '@inbox/contracts';

const app = new Hono();

app.post('/api/messages', async (c) => {
  const body = SendRequestSchema.parse(
    await c.req.json()
  );

  const result = await inbox.send({
    from: c.get('actor'),
    to: body.to,
    cc: body.cc,
    subject: body.subject,
    body: body.body,
  });

  return c.json({ ok: true, data: result });
});


app.get('/api/messages/:id', async (c) => {
  const id = MessageIdSchema.parse(c.req.param('id'));
  const message = await inbox.read(id, c.get('actor'));
  return c.json({ ok: true, data: message });
});


app.get('/api/inbox', async (c) => {
  const query = InboxQuerySchema.parse(c.req.query());
  const items = await inbox.list(c.get('actor'), query);
  return c.json({ ok: true, data: items });
});


app.post('/api/messages/:id/ack', async (c) => {
  const id = MessageIdSchema.parse(c.req.param('id'));
  const result = await inbox.ack(id, c.get('actor'));
  return c.json({ ok: true, data: result });
});

import { useMutation } from '@tanstack/react-query';
import type { SendRequest } from '@inbox/contracts';

function ComposeForm() {
  const send = useMutation({
    mutationFn: (data: SendRequest) =>
      fetch('/api/messages', {
        method: 'POST',
        body: JSON.stringify(data),
      }).then(r => r.json()),
  });

  return (
    <form onSubmit={(e) => {
      e.preventDefault();
      send.mutate(formData);
    }}>
      {/* form fields */}
    </form>
  );
}


function MessageReader({ id }: { id: MessageId }) {
  const { data } = useQuery({
    queryKey: ['message', id],
    queryFn: () => fetch(`/api/messages/${id}`)
      .then(r => r.json()),
  });

  return <MessageView message={data} />;
}


function InboxList() {
  const { data } = useQuery({
    queryKey: ['inbox'],
    queryFn: () => fetch('/api/inbox').then(r => r.json()),
  });

  return (
    <ul>
      {data?.map(item => (
        <InboxRow key={item.delivery_id} item={item} />
      ))}
    </ul>
  );
}


function AckButton({ id }: { id: MessageId }) {
  const ack = useMutation({
    mutationFn: () =>
      fetch(`/api/messages/${id}/ack`, { method: 'POST' })
        .then(r => r.json()),
  });

  return (
    <button onClick={() => ack.mutate()}
      disabled={ack.isPending}>
      {ack.isPending ? 'Acknowledging...' : 'Acknowledge'}
    </button>
  );
}

INSTALLATION

          
One command. Zero dependencies.

          
No Docker. No cloud. No runtime. Just bash and sqlite3.

        


        

          $ curl -fsSL https://tryinbox.sh/install | bash
          
        


        

          npm i @inbox/contracts  TypeScript types only
          

          brew install inbox  coming soon
        


        

          

            

              
            

            
SQLite-backed

            
Single file database. WAL mode. STRICT tables. No ORM, no connection pool, no cluster. Your data is a file you can copy.

          


          

            

              
            

            
Immutable messages

            
Messages are write-once. No edits, no deletes. Recipients get local deliveries they control. The shared truth never changes.

          


          

            

              
            

            
Conversation lineage

            
Conversations are containers, not ACL boundaries. Visibility is strictly delivery-based. Lineage does not grant access.

          


          

            

              
            

            
Idempotent mutations

            
Read, ack, hide, unhide. Every state change is idempotent. No-ops return changed: false and append no events. Safe to retry.

          


          

            

              
            

            
Lists with fan-out

            
Address groups expand at send time. Inactive members are silently skipped. Reply-all uses logical headers, not expanded snapshots.

          


          

            

              
            

            
JSON-native

            
Every command supports --json. Flat response envelopes. Typed error codes. Stable contracts. Pipe to jq, parse in any language.

          

        


        

          Requirements: bash 4+, sqlite3 3.35+. That's it.
